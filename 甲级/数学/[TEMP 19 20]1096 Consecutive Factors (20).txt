#include <vector>
#include <iostream>
#include <algorithm>
#include <cmath>
#include <climits>
using namespace std;

long long int cres(int a, int b) {
	long long int res = 1;
	for (int i = a; i <= b; i++) {
		res *= i;
		if (res > INT_MAX) {
			res = INT_MAX + 1;
			break;
		}
	}
	return res;
}

int main() {
	int n, min_n = INT_MAX, max_l = 0;
	long long int cr;
	cin >> n;
	bool flag = 0;
	bool prime = 1;
	for (int i = 0; ; i++) {
		for (int j = 2; j <= sqrt(n); j++) {
			cr = cres(j, j + i);
			if (j == 2 && cr > n) {
				flag = 1;
				break;
			}
			if (cr > n) break;
			if (n%cr == 0) {
				if (i == 0) prime = 0;
				if (i + 1 > max_l) {
					max_l = i + 1;
					min_n = j;
				}
				break;
			}
		}
		if (prime == 1) {
			max_l = 1;
			min_n = n;
			break;
		}
		if (flag == 1) break;
	}
	cout << max_l << endl;
	for (int i = min_n; i < min_n + max_l; i++) {
		if (i != min_n) cout << "*";
		cout << i;
	}
	cout << endl;
	return 0;
}

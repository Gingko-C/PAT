#include <iostream>
using std::cout;
using std::cin;
using std::endl;
#include <vector>
#include <map>
#include <queue>
using std::priority_queue;
using std::map;
using std::vector;

#define INT_MAX 1000000

struct Node {
	bool visit = 0;
	int d = 0;
	int pi = -1;
	bool child = 0;
};

int main() {
	int n, m;
	cin >> n >> m;
	vector<Node> vn(n + 1);
	map<int, vector<int>> f2c;
	int fname, cnum, cname;
	vector<int> dncnum(m);
	for (int i = 0; i < m; i++) {
		cin >> fname >> cnum;
		vn[fname].child = 1;
		for (int j = 0; j < cnum; j++) {
			cin >> cname;
			vn[cname].pi = fname;
			f2c[fname].push_back(cname);
		}
	}
	//int root;
	//for (int i = 1; i <= n; i++) {
	//	if (vn[i].pi = -1) {
	//		root = i;
	//		break;
	//	}
	//}
	priority_queue<int> pqi;
	pqi.push(1);
	while (!pqi.empty()) {
		int temp_node = pqi.top();
		pqi.pop();
		vn[temp_node].visit = 1;
		for (auto &c : f2c[temp_node]) {
			vn[c].d = vn[temp_node].d + 1;
			vn[c].visit = 1;
			if (vn[c].child == 0)
				dncnum[vn[c].d]++;
		}
	}
	system("pause");
	return 0;
}
